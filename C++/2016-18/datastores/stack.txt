Last In, First Out

functions: push, pop, stack_empty

1.push(s, x): // s is the bottom marker, x is the input
  top[s] <- top[s] + 1;
  s[top[s]] <- x

2.pop(s):
  if (stack_empty)
    return s;
  else top[s] <- top[s] - 1;
    return s[top[s]+1]

3.stack_empty:
  if top[x] = 0
    return true;
  else return false;
